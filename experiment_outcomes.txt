def build_prompt(question: str) -> str:
    question = question.strip()

    if not question:
        raise ValueError("question must not be empty.")

    return (
        "Answer the following question concisely.\n\n"
        f"Question:\n{question}\n\n"
        "Answer:\n"
    )

from dataclasses import dataclass

@dataclass(frozen=True)
class ScoredDevelopmentQuery:
    example_id: str
    question: str
    prompt: str
    draft: str
    uncertainty_score: float

scored_queries: list[ScoredDevelopmentQuery] = []

for index, question in enumerate(development_questions):
    
    prompt = build_prompt(question)

    draft = draft_generator.generate(prompt)

    uncertainty = scorer.score(draft)

    scored_queries.append(
        ScoredDevelopmentQuery(
            example_id=f"dev_{index:03d}",
            question=question,
            prompt=prompt,
            draft=draft.generated_text,
            uncertainty_score=float(uncertainty.score),
        )
    )

    print(
        f"{index + 1:02d}/{len(development_questions)} | "
        f"U={uncertainty.score:.6f} | "
        f"{question}"
    )

def run_query(query: str):

    print("=" * 80)
    print(f"Query: {query}")
    print("=" * 80)

    draft = draft_generator.generate(query)

    print("\nDraft:")
    print(draft.generated_text)

    uncertainty = scorer.score(draft)

    print("\nUncertainty Score:")
    print(f"{uncertainty.score:.6f}")

    decision = retrieval_gate.decide(
        uncertainty.score
    )

    print("\nGate Decision:")
    print(decision.retrieve)

    print(f"Threshold : {decision.threshold:.6f}")

    if decision.retrieve:

        retrieval = retriever.retrieve(
            query,
            top_k=5
        )
    else:

      retrieval = RetrievalResult(
          query=query,
          documents=[],
          retrieved=False,
          retrieval_backend="None"
      )

    print("\nRetrieved Documents:")

    print(len(retrieval.documents))

    for i, doc in enumerate(retrieval.documents):

        print("-" * 60)

        print(f"Document {i+1}")

        print(f"ID: {doc.id}")

        print(f"Score: {doc.score}")

        print(doc.text[:250])

    return {
        "draft": draft,
        "uncertainty": uncertainty,
        "decision": decision,
        "retrieval": retrieval,
    }
